+    {
+        RHI_Texture* tex_debug = GetRenderTarget(Renderer_RenderTarget::debug_output);

+        }
+
+        RHI_Texture* tex_depth = GetRenderTarget(Renderer_RenderTarget::gbuffer_depth);
